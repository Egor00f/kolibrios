248__attribute__((__stdcall__)) static inline void filebrowser_key(filebrowser *fb, oskey_t keypress)
250{
251// if (!fb->select_flag) return; // same reaction as other controls
252
253 int extended_key = 0, act = 0;
254
255 if(keypress.state) return;
256 if (keypress.code == 0xE0){ extended_key = 1; return; }
257
258 act = 0;
259 switch(keypress.ctrl_key) // ascii scancode
260 {
261 case 80: // arrow down
262 act = 1; break;
263 case 72: // arrow up
264 act = 2; break;
265 case 81: // PageDown
266 act = 3; break;
267 case 73: // PageUp
268 act = 4; break;
269 case 71: // Home
270 act = 5; break;
271 case 79: // End
272 act = 6; break;
273 case 28: // Enter
274 act = 7; break;
275 case 82: // Insert
276 act = 8; break;
277 case 78: // NumPad+ select all
278 act = 9; break;
279 case 74: // NumPad- deselct
280 act = 10; break;
281 case 55: // NumPad* invert selection
282 act = 11; break;
283 default:
284 act = 12; // search by letter
285 }
286 fb->key_action = act;
287 fb->key_action_num = keypress.ctrl_key;
288
289// debug_board_printf("key pressed [%X] %d, action %d, ext_flag = %d\n", keypress.val, brows.key_action_num, act, extended_key);

291 if (extended_key) extended_key = 0;
292 (*filebrowse_key)(fb);
293}
