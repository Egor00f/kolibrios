64/* Initializes an Editbox with sane settings, sufficient for most use.
65 This will let you create a box and position it somewhere on the screen.
66 The text_buffer is a pointer to a character array and needs to be as long as
67 AT LEAST MAX_CHARS + 1.If the text_buffer is smaller, it will crash if user
68 types more characters than what will fit into the text buffer.
69
70 Allocating buffer space automatically so that programmer can be carefree now.
71 This also automatically adjusts the size of edit box so that it can hold enough characters.
72
73 All you need is :
74
75 tlx,tly = Coordinates of the beginning of the edit box.
76 max_chars = Limit of number of characters user can enter into edit box.

79edit_box* kolibri_new_edit_box(unsigned int tlx, unsigned int tly, unsigned int max_chars, void *editbox_interlock)
80{
81 unsigned int PIXELS_PER_CHAR = 7;
82 edit_box *new_textbox = (edit_box *)calloc(1, sizeof(edit_box));
83 char *text_buffer = (char *)calloc(max_chars + 2, sizeof(char)); // +2 as asked in box_lib src
84
85 /* Update blur_border_color and shift_color from box_lib.mac macro */
86 /* edit_boxes_set_sys_color */
87
88 new_textbox -> width = max_chars * PIXELS_PER_CHAR;
89 new_textbox -> left = tlx;
90 new_textbox -> top = tly;
91 new_textbox -> color = 0xFFFFFF; /* Always make white edit boxes */
92 new_textbox -> shift_color = 0x6a9480;
93 new_textbox -> focus_border_color = kolibri_color_table.color_work_graph;
94 new_textbox -> blur_border_color = 0x6a9480;
95 new_textbox -> text_color = kolibri_color_table.color_work_text; /* Always black text when typing */
96 new_textbox -> max = max_chars;
97 new_textbox -> text = text_buffer;
98 new_textbox -> mouse_variable = editbox_interlock;
99 new_textbox -> flags = 0x00000000;
100
101 return new_textbox;
102}
