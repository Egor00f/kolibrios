67/* Initializes an Editbox with sane settings, sufficient for most use.
68 This will let you create a box and position it somewhere on the screen.
69 The text_buffer is a pointer to a character array and needs to be as long as
70 AT LEAST MAX_CHARS + 1.If the text_buffer is smaller, it will crash if user
71 types more characters than what will fit into the text buffer.
72
73 Allocating buffer space automatically so that programmer can be carefree now.
74 This also automatically adjusts the size of edit box so that it can hold enough characters.
75
76 All you need is :
77
78 tlx,tly = Coordinates of the beginning of the edit box.
79 max_chars = Limit of number of characters user can enter into edit box.
80*/
81
82edit_box* kolibri_new_edit_box(unsigned int tlx, unsigned int tly, unsigned int max_chars, void *editbox_interlock)
83{
84 unsigned int PIXELS_PER_CHAR = 7;
85 edit_box *new_textbox = (edit_box *)calloc(1, sizeof(edit_box));
86 char *text_buffer = (char *)calloc(max_chars + 2, sizeof(char)); // +2 as asked in box_lib src
87
88 /* Update blur_border_color and shift_color from box_lib.mac macro */
89 /* edit_boxes_set_sys_color */
90
91 new_textbox -> width = max_chars * PIXELS_PER_CHAR;
92 new_textbox -> left = tlx;
93 new_textbox -> top = tly;
94 new_textbox -> color = 0xFFFFFF; /* Always make white edit boxes */
95 new_textbox -> shift_color = 0x6a9480;
96 new_textbox -> focus_border_color = kolibri_color_table.color_work_graph;
97 new_textbox -> blur_border_color = 0x6a9480;
98 new_textbox -> text_color = kolibri_color_table.color_work_text; /* Always black text when typing */
99 new_textbox -> max = max_chars;
100 new_textbox -> text = text_buffer;
101 new_textbox -> mouse_variable = editbox_interlock;
102 new_textbox -> flags = 0x00000000;
103
104 return new_textbox;
105}
